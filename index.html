function renderFinal(r, w) { setSc('sc-result'); const container = document.getElementById('result-container'); container.innerHTML = ""; // 1. 도보 1분 컷 필터링 const paths = r.result?.path?.filter(p => p.subPath[0].trafficType !== 3 || p.subPath[0].sectionTime <= 1) || []; // 2. 지하철/버스 최적 경로 추출 const subway = paths.filter(p => p.pathType === 1).sort((a,b) => a.info.totalTime - b.info.totalTime)[0]; const bus = paths.filter(p => p.pathType !== 1).sort((a,b) => a.info.totalTime - b.info.totalTime)[0]; let displayList = []; if (subway && bus) { const diff = subway.info.totalTime - bus.info.totalTime; if (diff <= 0) displayList = [subway]; else if (diff < 15) displayList = [subway, bus]; else displayList = [bus]; } else { if (subway) displayList.push(subway); else if (bus) displayList.push(bus); } displayList.forEach((p, idx) => { const isBus = p.pathType !== 1; const margin = isBus ? 15 : 10; // [핵심: 실제 배차 시각 추출] // 오디세이는 '도착시간 기준' 조회 시 p.info.firstStartStationPointTime에 실제 출발 시각을 줍니다. // 만약 해당 필드가 없으면 이론적 최후 시각에서 마진을 뺍니다. let actualDeparture; if (p.info.firstStartStationPointTime) { // API가 주는 "13:05" 형태의 문자열을 파싱 const [h, m] = p.info.firstStartStationPointTime.split(':').map(Number); actualDeparture = new Date(); actualDeparture.setHours(h, m - margin); // 실제 차 시간에서 마진 뺌 } else { // 데이터가 없을 경우 보수적으로 소요시간 + 마진 역산 actualDeparture = new Date(); actualDeparture.setHours(app.time, -(p.info.totalTime + margin)); } const startPoint = isBus ? POS.BUS : POS.SUB; const card = document.createElement('div'); card.className = 'card'; if(idx === 0 && displayList.length > 1) card.style.borderTop = "5px solid var(--blue)"; let html = `${isBus ? '🚌 버스/복합 (마진 15분)' : '🚇 지하철 전용 (마진 10분)'}`; html += `
${String(actualDeparture.getHours()).padStart(2,'0')}:${String(actualDeparture.getMinutes()).padStart(2,'0')}
`; // 배차 정보가 있을 경우 추가 메시지 if (p.info.firstStartStationPointTime) { html += `
${p.info.firstStartStationPointTime} 차량 기준 역산됨
`; } html += `
${startPoint.name} 출발 권장
`; html += `
` + p.subPath.filter(s => s.trafficType !== 3).map((s, i) => `
${s.trafficType===1?s.lane[0].name:s.lane[0].busNo} ${i===0?startPoint.name:s.startName} 탑승 ➔ ${s.endName}
`).join('
▼ 환승
') + `
`; card.innerHTML = html; container.appendChild(card); }); renderWeather(w); }
